var p = function(n) {
    var p,i,
    size = n /2-1,
    d= new Array(size),
    toplam = 2;

    for (i = 0,p = 3; p< n; p+=2,i++)d [i] = p;

    for (p = 0; p  <size;p++){
        if (d[p]===0) continue;
        toplam += d[p];
        for (i = p +d[p]; i < size; i += d[p]) d[i] =0;
    





    }
    return toplam;
};
